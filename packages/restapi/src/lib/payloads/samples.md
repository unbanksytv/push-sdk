async function trigger(input, name) {
    console.log(`........${name}...starts............>>\n`);
    try {
        const apiResponse = await sendNotification(input);
        console.log('apiResponse: ', apiResponse.status);
    } finally {
        console.log(`<<........${name}....ends...........\n`);
    }
}
